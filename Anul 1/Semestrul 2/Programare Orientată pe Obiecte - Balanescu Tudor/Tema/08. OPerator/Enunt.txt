#include <iostream.h>

class C{
    public:
        C(int i=0){x=i;}
        C& operator++(){++x; return *this;}
        C  operator--(){--x; return *this;}
    private:
        int x;
};

void main(){
    C i;
    cout<<i<<endl;
    cout<<++(++i)<<endl<<i<<endl;
    cout<<--(--i)<<endl<<i<<endl;
}
